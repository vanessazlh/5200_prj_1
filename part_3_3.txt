SELECT
    area_name,
    SUM(CASE WHEN acceptance = 'accepted' THEN 1 ELSE 0 END) AS accepted_papers,
    SUM(CASE WHEN acceptance = 'rejected' THEN 1 ELSE 0 END) AS rejected_papers,
    SUM(CASE WHEN acceptance = 'undecided' THEN 1 ELSE 0 END) AS undecided_papers
FROM
    Paper
GROUP BY
    area_name
ORDER BY
    area_name;
