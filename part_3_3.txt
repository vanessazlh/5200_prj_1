SELECT 
    p.area_name AS research_area,
    SUM(CASE WHEN p.acceptance = 'accepted' THEN 1 ELSE 0 END) AS accepted_count,
    SUM(CASE WHEN p.acceptance = 'rejected' THEN 1 ELSE 0 END) AS rejected_count
FROM 
    Paper p
JOIN 
    Research_Area r ON p.area_name = r.area_name
GROUP BY 
    r.area_name
ORDER BY 
    r.area_name;
