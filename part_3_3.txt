SELECT
    ra.area_name,
    COUNT(CASE WHEN p.acceptance = 'accepted' THEN 1 END) AS accepted_papers,
    COUNT(CASE WHEN p.acceptance = 'rejected' THEN 1 END) AS rejected_papers,
    COUNT(CASE WHEN p.acceptance = 'undecided' THEN 1 END) AS undecided_papers,
    COUNT(p.pid) AS total_papers
FROM
    Research_Area ra
LEFT JOIN Paper p ON ra.area_name = p.area_name
GROUP BY
    ra.area_name
ORDER BY
    ra.area_name;
